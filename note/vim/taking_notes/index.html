Simple notes inside the terminal

June 8, 2020
vim vimwiki

Vimwiki allows to create notes in a terminal.
Remember to configure vim (.vimrc):

let g:vimwiki_list = [{'path': 'path_to_notes/',
                      \ 'syntax': 'markdown', 'ext': '.md'}]

For easier navigation customize vim browser source

let g:netrw_banner = 0
let g:netrw_liststyle = 3
let g:netrw_browse_split = 4
let g:netrw_altv = 1
let g:netrw_winsize = 20
augroup ProjectDrawer
  autocmd!
  autocmd VimEnter * :Vexplore
augroup END
